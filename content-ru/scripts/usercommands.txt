function main(...)
	if arg.n == 0 then 
		client:send("no arg");
	else
		client:send("Аргументов: "..#arg);
		for i = 1, #arg do
			client:send(arg[i]);
		end
	end
end

function help(...)
	client:send("Команда: тест\n\tЭто тестовый эхо-скрипт для проверки аргументов.");
end

##### справка #####
function main(...)
	if #arg == 0 then
		scripts = client:listScripts();
		client:send("Список доступных команд: "..scripts);
	else
		cmd = client:getFullCommand(arg[1]);
		
		if cmd ~= nil then
			client:doFunction(cmd,"help");
		else
			client:send("Справки по этой команде пока нет.");
		end
	end
end

function help(...)
	client:send([[Команда: справка
	Синтаксис: "справка"  или "справка" <имя команды>
	- Без аргументов выводит список скриптов.
	- С одним аргументом выводит справку об указанном в аргументе скрипте.]]);
end
